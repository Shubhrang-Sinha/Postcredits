```typescript
'use client';

import { Button } from '@radix-ui/themes';

interface Genre {
  genre_id: number;
  name: string;
}

interface GenreFilterProps {
  genres: Genre[];
  selectedGenre: number | null;
  onGenreSelect: (genreId: number | null) => void;
}

export default function GenreFilter({ genres, selectedGenre, onGenreSelect }: GenreFilterProps) {
  return (
    <div className="flex flex-wrap gap-2 mb-6">
      <Button
        variant={selectedGenre === null ? 'solid' : 'outline'}
        size="1"
        onClick={() => onGenreSelect(null)}
      >
        All
      </Button>
      {genres.map((genre) => (
        <Button
          key={genre.genre_id}
          variant={selectedGenre === genre.genre_id ? 'solid' : 'outline'}
          size="1"
          onClick={() => onGenreSelect(genre.genre_id)}
        >
          {genre.name}
        </Button>
      ))}
    </div>
  );
}
```

```typescript
import Link from 'next/link';
import { Card, Text, Flex } from '@radix-ui/themes';

interface MediaCardProps {
  id: number;
  title: string;
  year?: number;
  rating?: number;
  type: 'movie' | 'book';
  creator?: string; // director for movies, author for books
  genres?: string[];
}

export default function MediaCard({ id, title, year, rating, type, creator, genres }: MediaCardProps) {
  const href = `/${type === 'movie' ? 'movies' : 'books'}/${id}`;

  return (
    <Link href={href}>
      <Card size="2" style={{ width: 200, cursor: 'pointer' }}>
        <Flex direction="column" gap="2">
          <div className="h-40 bg-gray-200 rounded flex items-center justify-center">
            <Text size="9" color="gray">🎬</Text>
          </div>
          <Text weight="bold" size="3" trim="start">
            {title}
          </Text>
          {year && (
            <Text size="2" color="gray">
              {year}
            </Text>
          )}
          {creator && (
            <Text size="2" color="gray">
              {creator}
            </Text>
          )}
          {rating && (
            <Text size="2">
              ⭐ {rating.toFixed(1)}
            </Text>
          )}
          {genres && genres.length > 0 && (
            <Text size="1" color="gray">
              {genres.slice(0, 2).join(', ')}
            </Text>
          )}
        </Flex>
      </Card>
    </Link>
  );
}
```

```typescript
import { Suspense } from 'react';
import api from '@/lib/api';
import GenreFilter from '@/components/GenreFilter';
import MediaCard from '@/components/MediaCard';
import { Flex, Heading, Spinner, Text } from '@radix-ui/themes';

interface Book {
  book_id: number;
  work_id: number;
  title: string;
  release_year: number;
  pages: number;
  author_name: string;
  average_rating: number;
  genres: string[];
}

interface Genre {
  genre_id: number;
  name: string;
}

async function getBooks(genreId?: number) {
  const url = genreId ? `/books?genre=${genreId}` : '/books';
  const result = await api.get<Book[]>(url);
  return result.data || [];
}

async function getGenres() {
  const result = await api.get<Genre[]>('/genres');
  return result.data || [];
}

export default async function BooksPage({
  searchParams,
}: {
  searchParams: { genre?: string };
}) {
  const genreId = searchParams.genre ? parseInt(searchParams.genre) : undefined;

  const [books, genres] = await Promise.all([
    getBooks(genreId),
    getGenres(),
  ]);

  return (
    <main style={{ maxWidth: 1200, margin: '0 auto', padding: '20px' }}>
      <Heading size="8" mb="4">Books</Heading>

      <GenreFilter
        genres={genres.filter(g => g.name.toLowerCase() !== 'movie')}
        selectedGenre={genreId || null}
        onGenreSelect={(id) => {
          // This would need client component for dynamic navigation
          window.location.href = id ? `/books?genre=${id}` : '/books';
        }}
      />

      {books.length === 0 ? (
        <Text>No books found</Text>
      ) : (
        <Flex gap="4" wrap="wrap">
          {books.map((book) => (
            <MediaCard
              key={book.book_id}
              id={book.work_id}
              title={book.title}
              year={book.release_year}
              rating={book.average_rating}
              type="book"
              creator={book.author_name}
              genres={book.genres}
            />
          ))}
        </Flex>
      )}
    </main>
  );
}
```
