```typescript
'use client';

import { useState } from 'react';
import { Button, Flex, Text } from '@radix-ui/themes';

interface StarRatingProps {
  value: number; // 0-5
  onChange?: (value: number) => void;
  readonly?: boolean;
  size?: 'small' | 'medium' | 'large';
}

export default function StarRating({ value, onChange, readonly = false, size = 'medium' }: StarRatingProps) {
  const [hoverValue, setHoverValue] = useState(0);

  const displayValue = hoverValue || value;

  const starSize = {
    small: '12px',
    medium: '24px',
    large: '36px',
  }[size];

  return (
    <Flex gap="1">
      {[1, 2, 3, 4, 5].map((star) => (
        <button
          key={star}
          type="button"
          disabled={readonly}
          onClick={() => !readonly && onChange?.(star)}
          onMouseEnter={() => !readonly && setHoverValue(star)}
          onMouseLeave={() => !readonly && setHoverValue(0)}
          style={{
            background: 'none',
            border: 'none',
            cursor: readonly ? 'default' : 'pointer',
            fontSize: starSize,
            padding: '2px',
          }}
        >
          {star <= displayValue ? '⭐' : '☆'}
        </button>
      ))}
    </Flex>
  );
}
```

```typescript
import { notFound } from 'next/navigation';
import api from '@/lib/api';
import StarRating from '@/components/StarRating';
import { Button, Card, Flex, Heading, Text, Separator } from '@radix-ui/themes';
import BookRatingClient from './BookRatingClient';

interface Book {
  book_id: number;
  work_id: number;
  title: string;
  release_year: number;
  pages: number;
  author_name: string;
  author_id: number;
  average_rating: number;
  genres: string[];
}

interface Rating {
  rating_id: number;
  score: number;
}

async function getBook(id: string): Promise<Book | null> {
  const result = await api.get<Book>(`/books/${id}`);
  return result.data || null;
}

async function getUserRating(workId: number): Promise<Rating | null> {
  const result = await api.get<Rating[]>(`/ratings?workId=${workId}`);
  return result.data?.[0] || null;
}

export default async function BookDetailPage({ params }: { params: { id: string } }) {
  const book = await getBook(params.id);

  if (!book) {
    notFound();
  }

  const userRating = await getUserRating(book.work_id);

  return (
    <main style={{ maxWidth: 800, margin: '0 auto', padding: '20px' }}>
      <Card size="4">
        <Flex direction="column" gap="4">
          <Heading size="9">{book.title}</Heading>

          <Flex gap="4">
            <Text size="3">Author: {book.author_name}</Text>
            <Text size="3">Year: {book.release_year}</Text>
            <Text size="3">Pages: {book.pages}</Text>
          </Flex>

          {book.genres.length > 0 && (
            <Flex gap="2">
              {book.genres.map((genre) => (
                <Text key={genre} size="2" color="gray">
                  {genre}
                </Text>
              ))}
            </Flex>
          )}

          <Flex justify="between" align="center">

          </Flex>

        </Flex>
      </Card>
    </main>
  );
}
```

Create frontend/app/books/[id]/BookRatingClient.tsx:

```typescript
'use client';

import { useState } from 'react';
import { useAuth } from '@/lib/auth-context';
import api from '@/lib/api';
import StarRating from '@/components/StarRating';
import { Button, Card, Flex, Text } from '@radix-ui/themes';

interface BookRatingClientProps {
  workId: number;
  initialRating?: number;
}

export default function BookRatingClient({ workId, initialRating }: BookRatingClientProps) {
  const { isAuthenticated } = useAuth();
  const [userRating, setUserRating] = useState(initialRating || 0);
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [message, setMessage] = useState('');

  const handleRating = async (rating: number) => {
    if (!isAuthenticated) {
      setMessage('Please login to rate');
      return;
    }

    setIsSubmitting(true);

    if (userRating > 0) {
      // Update existing rating
      const result = await api.put(`/ratings/${workId}`, { score: rating });
      if (result.error) {
        setMessage('Failed to update rating');
      } else {
        setUserRating(rating);
        setMessage('Rating updated!');
      }
    } else {
      // Create new rating
      const result = await api.post('/ratings', { workId, score: rating });
      if (result.error) {
        setMessage('Failed to submit rating');
      } else {
        setUserRating(rating);
        setMessage('Rating submitted!');
      }
    }

    setIsSubmitting(false);

    // Clear message after 3 seconds
    setTimeout(() => setMessage(''), 3000);
  };

  if (!isAuthenticated) {
    return (
      <Card>
        <Text>Please login to rate this book</Text>
      </Card>
    );
  }

  return (
    <Card>
      <Flex direction="column" gap="3">
        <Text size="3" weight="bold">Your Rating:</Text>

        <StarRating
          value={userRating}
          onChange={handleRating}
          readonly={isSubmitting}
        />

        {message && (
          <Text size="2" color={message.includes('Failed') ? 'red' : 'green'}>
            {message}
          </Text>
        )}
      </Flex>
    </Card>
  );
}
```
