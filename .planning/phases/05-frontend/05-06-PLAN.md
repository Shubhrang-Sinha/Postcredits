```typescript
'use client';

import { Flex, Text } from '@radix-ui/themes';

interface StatsChartProps {
  title: string;
  data: { label: string; value: number }[];
  color?: string;
}

export default function StatsChart({ title, data, color = '#06b6d4' }: StatsChartProps) {
  if (data.length === 0) {
    return (
      <div className="p-4 border rounded">
        <Text size="4" weight="bold">{title}</Text>
        <Text size="2" color="gray" className="mt-2">No data available</Text>
      </div>
    );
  }

  const maxValue = Math.max(...data.map(d => d.value));

  return (
    <div className="p-4 border rounded">
      <Text size="4" weight="bold" mb="4" className="block">{title}</Text>

      <div className="space-y-2">
        {data.slice(0, 10).map((item, index) => (
          <div key={index} className="flex items-center gap-2">
            <Text size="2" className="w-24 truncate">{item.label}</Text>
            <div
              className="h-6 rounded"
              style={{
                width: `${(item.value / maxValue) * 200}px`,
                backgroundColor: color,
                minWidth: item.value > 0 ? '4px' : '0'
              }}
            />
            <Text size="2" color="gray">{item.value}</Text>
          </div>
        ))}
      </div>
    </div>
  );
}
```

```typescript
import { redirect } from 'next/navigation';
import api from '@/lib/api';
import { useAuth } from '@/lib/auth-context';
import StatsChart from '@/components/StatsChart';
import { Flex, Heading, Text, Tabs } from '@radix-ui/themes';

// Auth check wrapper
function StatsContent() {
  const { isAuthenticated } = useAuth();

  if (!isAuthenticated) {
    redirect('/login');
  }

  return <StatsDashboard />;
}

interface GenreStat {
  genre: string;
  count: number;
}

interface YearStat {
  year: number;
  count: number;
  avg_rating: number;
}

async function StatsDashboard() {
  // Fetch all stats in parallel
  const [movieGenresResult, bookGenresResult, movieYearsResult, bookYearsResult] = await Promise.all([
    api.get<GenreStat[]>('/stats/genres?type=movie'),
    api.get<GenreStat[]>('/stats/genres?type=book'),
    api.get<YearStat[]>('/stats/years?type=movie'),
    api.get<YearStat[]>('/stats/years?type=book'),
  ]);

  const movieGenres = movieGenresResult.data || [];
  const bookGenres = bookGenresResult.data || [];
  const movieYears = movieYearsResult.data || [];
  const bookYears = bookYearsResult.data || [];

  const movieGenreData = movieGenres.map(g => ({ label: g.genre, value: g.count }));
  const bookGenreData = bookGenres.map(g => ({ label: g.genre, value: g.count }));
  const movieYearData = movieYears
    .filter(y => y.count > 0)
    .map(y => ({ label: y.year.toString(), value: y.count }));
  const bookYearData = bookYears
    .filter(y => y.count > 0)
    .map(y => ({ label: y.year.toString(), value: y.count }));

  return (
    <main style={{ maxWidth: 1200, margin: '0 auto', padding: '20px' }}>
      <Heading size="8" mb="2">Spotistats</Heading>
      <Text size="3" color="gray" mb="6">Your viewing and reading statistics</Text>

      <Tabs.Root defaultValue="movies">
        <Tabs.List>
          <Tabs.Trigger value="movies">Movies</Tabs.Trigger>
          <Tabs.Trigger value="books">Books</Tabs.Trigger>
        </Tabs.List>

        <Tabs.Content value="movies">
          <Flex direction="column" gap="6" mt="6">
            <StatsChart
              title="Movies by Genre"
              data={movieGenreData}
              color="#06b6d4" // cyan
            />
            <StatsChart
              title="Movies by Year"
              data={movieYearData}
              color="#8b5cf6" // violet
            />
          </Flex>
        </Tabs.Content>

        <Tabs.Content value="books">
          <Flex direction="column" gap="6" mt="6">
            <StatsChart
              title="Books by Genre"
              data={bookGenreData}
              color="#10b981" // emerald
            />
            <StatsChart
              title="Books by Year"
              data={bookYearData}
              color="#f59e0b" // amber
            />
          </Flex>
        </Tabs.Content>
      </Tabs.Root>
    </main>
  );
}

export default StatsContent;
```
