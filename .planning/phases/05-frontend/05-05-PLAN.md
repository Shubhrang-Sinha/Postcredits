```typescript
'use client';

import { useState } from 'react';
import { Button, Flex, Text } from '@radix-ui/themes';
import MediaCard from './MediaCard';

interface Recommendation {
  work_id: number;
  title: string;
  work_type: string;
  average_rating: number;
  release_year?: number;
  similarity?: number;
}

interface RecommendationCarouselProps {
  title: string;
  recommendations: Recommendation[];
}

export default function RecommendationCarousel({ title, recommendations }: RecommendationCarouselProps) {
  const [index, setIndex] = useState(0);
  const itemsPerPage = 4;
  
  const totalPages = Math.ceil(recommendations.length / itemsPerPage);
  const currentItems = recommendations.slice(index * itemsPerPage, (index + 1) * itemsPerPage);
  
  if (recommendations.length === 0) {
    return null;
  }

  return (
    <div className="mb-8">
      <Flex justify="between" align="center" mb="4">
        <Text size="5" weight="bold">{title}</Text>
        <Flex gap="2">
          <Button 
            variant="outline" 
            size="1"
            onClick={() => setIndex(Math.max(0, index - 1))}
            disabled={index === 0}
          >
            ←
          </Button>
          <Text size="2" className="flex items-center">
            {index + 1} / {totalPages}
          </Text>
          <Button 
            variant="outline" 
            size="1"
            onClick={() => setIndex(Math.min(totalPages - 1, index + 1))}
            disabled={index >= totalPages - 1}
          >
            →
          </Button>
        </Flex>
      </Flex>
      
      <Flex gap="4" wrap="wrap">
        {currentItems.map((rec) => (
          <MediaCard
            key={rec.work_id}
            id={rec.work_id}
            title={rec.title}
            year={rec.release_year}
            rating={rec.average_rating}
            type={rec.work_type as 'movie' | 'book'}
          />
        ))}
      </Flex>
    </div>
  );
}
```

```typescript
import { redirect } from 'next/navigation';
import api from '@/lib/api';
import { useAuth } from '@/lib/auth-context';
import MediaCard from '@/components/MediaCard';
import { Flex, Heading, Text } from '@radix-ui/themes';

// This is a client component wrapper that checks auth
function WatchlistContent() {
  const { isAuthenticated, user } = useAuth();
  
  if (!isAuthenticated) {
    redirect('/login');
  }
  
  return <WatchlistInner userId={user?.userId} />;
}

async function WatchlistInner({ userId }: { userId?: number }) {
  if (!userId) {
    return null;
  }
  
  const result = await api.get<any[]>('/ratings');
  const ratings = result.data || [];
  
  if (ratings.length === 0) {
    return (
      <main style={{ maxWidth: 1200, margin: '0 auto', padding: '20px' }}>
        <Heading size="8" mb="4">Watchlist</Heading>
        <Text>You haven't rated any items yet. Start browsing to add to your watchlist!</Text>
      </main>
    );
  }

  return (
    <main style={{ maxWidth: 1200, margin: '0 auto', padding: '20px' }}>
      <Heading size="8" mb="4">Watchlist</Heading>
      <Text size="3" mb="6" color="gray">
        {ratings.length} rated items
      </Text>
      
      <Flex gap="4" wrap="wrap">
        {ratings.map((rating: any) => (
          <MediaCard
            key={rating.rating_id}
            id={rating.work_id}
            title={rating.title}
            rating={rating.score}
            type={rating.work_type === 'movie' ? 'movie' : 'book'}
          />
        ))}
      </Flex>
    </main>
  );
}

// Re-export for use in server component
export default WatchlistContent;
```

```typescript
import api from '@/lib/api';
import RecommendationCarousel from '@/components/RecommendationCarousel';
import MediaCard from '@/components/MediaCard';
import { useAuth } from '@/lib/auth-context';
import { Flex, Heading, Text, Card, Button } from '@radix-ui/themes';
import Link from 'next/link';

// Client component for auth-dependent content
function AuthenticatedHome() {
  const { isAuthenticated } = useAuth();
  
  if (!isAuthenticated) {
    return <PublicHome />;
  }
  
  return <PersonalizedHome />;
}

async function PublicHome() {
  // Fetch trending movies for unauthenticated users
  const moviesResult = await api.get<any[]>('/movies?limit=6');
  const movies = moviesResult.data || [];
  
  const booksResult = await api.get<any[]>('/books?limit=6');
  const books = booksResult.data || [];
  
  return (
    <main style={{ maxWidth: 1200, margin: '0 auto', padding: '20px' }}>
      <Card size="4" className="mb-8">
        <Flex direction="column" gap="4" align="center" py="8">
          <Heading size="9">Welcome to Postcredits</Heading>
          <Text size="4" color="gray">Discover your next favorite book or movie</Text>
          <Flex gap="4" mt="4">
            <Link href="/register">
              <Button size="3">Get Started</Button>
            </Link>
            <Link href="/login">
              <Button variant="outline" size="3">Login</Button>
            </Link>
          </Flex>
        </Flex>
      </Card>
      
      <Heading size="7" mb="4">Popular Movies</Heading>
      <Flex gap="4" wrap="wrap" mb="8">
        {movies.slice(0, 4).map((movie: any) => (
          <MediaCard
            key={movie.movie_id}
            id={movie.work_id}
            title={movie.title}
            year={movie.release_year}
            rating={movie.average_rating}
            type="movie"
            creator={movie.director_name}
            genres={movie.genres}
          />
        ))}
      </Flex>
      
      <Heading size="7" mb="4">Popular Books</Heading>
      <Flex gap="4" wrap="wrap">
        {books.slice(0, 4).map((book: any) => (

            key={book.book_id}

            type="book"

    </main>

}

async function PersonalizedHome() {
  // Fetch personalized recommendations
  const movieRecsResult = await api.get<any[]>('/recommendations/blend?type=movie&limit=8');
  const movieRecs = movieRecsResult.data || [];
  
  const bookRecsResult = await api.get<any[]>('/recommendations/blend?type=book&limit=8');
  const bookRecs = bookRecsResult.data || [];
  
  // Fetch trending for variety
  const moviesResult = await api.get<any[]>('/movies?limit=4');
  const trendingMovies = moviesResult.data || [];
  
  const booksResult = await api.get<any[]>('/books?limit=4');
  const trendingBooks = booksResult.data || [];
  
  return (
    <main style={{ maxWidth: 1200, margin: '0 auto', padding: '20px' }}>
      <Heading size="8" mb="6">For You</Heading>
      
      {movieRecs.length > 0 && (
        <RecommendationCarousel 
          title="Recommended Movies" 
          recommendations={movieRecs} 
        />
      )}
      
      {bookRecs.length > 0 && (
        <RecommendationCarousel 
          title="Recommended Books" 
          recommendations={bookRecs} 
        />
      )}
      
      <Heading size="7" mb="4">Trending Movies</Heading>
      <Flex gap="4" wrap="wrap" mb="8">
        {trendingMovies.map((movie: any) => (
          <MediaCard
            key={movie.movie_id}
            id={movie.work_id}
            title={movie.title}
            year={movie.release_year}
            rating={movie.average_rating}
            type="movie"
            creator={movie.director_name}
            genres={movie.genres}
          />
        ))}
      </Flex>
      
      <Heading size="7" mb="4">Trending Books</Heading>
      <Flex gap="4" wrap="wrap">
        {trendingBooks.map((book: any) => (
          <MediaCard
            key={book.book_id}
            id={book.work_id}
            title={book.title}
            year={book.release_year}
            rating={book.average_rating}
            type="book"
            creator={book.author_name}
            genres={book.genres}
          />
        ))}
      </Flex>
    </main>
  );
}

export default AuthenticatedHome;
```
